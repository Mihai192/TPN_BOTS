<h1>Cerința</h1>

<p>O pereche de cuvinte, unul cu număr par de litere, iar celălalt cu număr impar de litere, se numește <i>descentrată</i> dacă se poate obține cuvântul cu număr par de litere din celălalt, prin duplicarea caracterului din mijlocul acestuia.</p>

<p><strong>Exemplu:</strong> <br>
perechile <code>crezi</code> și <code>creezi</code>, respectiv <code>a</code> și <code>aa</code> sunt <i>descentrate</i>.</p>

<h1>Date de intrare</h1>

<p>Programul citește de la tastatură un text.</p>

<h1>Date de ieșire</h1>

<p>Programul va afișa pe ecran mesajul <code>DA</code>, dacă acesta conține cel puțin o pereche <i>descentrată</i>, sau mesajul <code>NU</code> în caz contrar.</p>

<h1>Restricții și precizări</h1>

<ul>
	<li>textul are cel mult <code>100</code> de caractere</li>
	<li>cuvintele sunt formate din litere mici ale alfabetului englez</li>
	<li>cuvintele sunt separate prin câte un spațiu</li>
</ul>

<h3>Explicație</h3>

<p>Perechea <em>descentrată</em> de cuvinte este <code>crezi</code> – <code>creezi</code>.</p>						
<pre class="EnlighterJSRAW" data-enlighter-language="cpp">#include <iostream>
using namespace std;

int nrcif(const char s[])
{
    int i = 0;
    int cnt=0;
    while(s[i]!='\0')
        cnt++ , i++;
    return cnt;
}

bool nr(int i ,const char s[])
{
    return ((int)s[i] >= 65 &amp;&amp; (int)s[i] <= 90) || ((int)s[i] >= 97 &amp;&amp; (int)s[i] <= 122);
}

void SplitText(const char s[], char cuv[100][30], int &amp;n)
{
    n = 0;
    int i = 0 , j=0;
    bool ok=false;
    bool fct= false;
    while(s[i]!='\0')
    {
        if(nr(i , s))
        {
            if(ok)
            {
                if(fct==false)
                n++ , ok = false;
                else
                    ok=false , fct = false;
            }
            cuv[n][j]=s[i];
            j++;
            if(!nr(i+1 , s))
                cuv[n][j]='\0';
        }
        else
        {
            if(i == 0)
                fct=true;
            j=0;
            ok = true;
        }
        i++;
    }
    n++;
}

int main()
{
    char s[100] , m[100][30];
    int n=0;
    cin.getline(s , 100);
    SplitText(s , m, n);
    bool pereche=false;
    for(int i = 0 ; i < n ; ++i)
    {
        for(int j = 0 ; j < n ; ++j)
        {
            if(nrcif(m[j])%2==1 &amp;&amp; nrcif(m[i])%2==0)
            {
                int poz = nrcif(m[j]) / 2;
                bool pd=true;
                int q = 0 , k = 0;
                while(m[i][q]!='\0')
                {
                    if(q==poz)
                        q++;
                    if(m[i][q]!=m[j][k])
                        pd=false;
                    q++ , k++;
                }
                if(pd)
                    pereche = true;
            }
        }
    }
    if(pereche)
        cout << "DA";
    else
        cout << "NU";
    return 0;
}</pre>
