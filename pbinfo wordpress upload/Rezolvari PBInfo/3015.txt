<h1>Date de intrare</h1>

<p>Fișierul de intrare <code>fibointerval.in</code> conține pe prima linie numerele <code>n</code> si <code>Q</code>, iar pe următoarele <code>Q</code> linii câte două numere <code>a</code> si <code>b</code> reprezentând query-urile.</p>

<h1>Date de ieșire</h1>

<p>Fișierul de ieșire <code>fibointerval.out</code> va conține pe fiecare linie, în ordine, rezultatul celor <code>Q</code> query-uri, cele <code>3</code> valori care trebuie afișate fiind separate prin câte un spațiu.</p>

<h1>Restricții și precizări</h1>

<ul>
	<li><code>1 ≤ n ≤ 1000</code></li>
	<li><code>1 ≤ Q ≤ 10000</code></li>
	<li><code>1 ≤ a ≤ b ≤ n</code></li>
</ul>

struct poz
{
    int n;
    char C[1001];

int n , k , a , b;

poz aduna(poz a , poz b)
{
    poz s = {0 , {0}};
    s.n = a.n;
    int t = 0;
    for(int i = 1 ; i <= a.n ; i++)
    {
        int c = a.C[i] + b.C[i] + t;
        s.C[i] = c % 10;
        t = c / 10;
    }
    if(t > 0)
    {
        s.n++;
        s.C[s.n] = t;
    }
    return s;
}


poz scade(poz a , poz b)
{
    int t = 0;
    for(int i = 1 ; i <= a.n ; i++)
    {
        int c = a.C[i] - b.C[i] + t;
        if(c < 0)
        {
            a.C[i] = 10 + c;
            t = -1;
        }
        else
        {
            a.C[i] = c;
            t = 0;
        }
    }
    while(a.C[a.n] == 0) a.n--;
    return a;
}

void afisare(poz a)
{
    for(int i = a.n ; i >= 1 ; i--)
        cout << (int)a.C[i];
}
