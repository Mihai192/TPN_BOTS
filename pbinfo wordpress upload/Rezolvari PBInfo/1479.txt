<p>Un număr natural în baza <code>10</code> se numește prețios dacă numărul de cifre ale sale din baza <code>2</code> este număr prim.</p>

<h1>Cerința</h1>

<p>Se dă un interval <code>[a,b]</code>.Determinați câte numere prețioase se află în acest interval.</p>

<h1>Date de intrare</h1>

<p>Programul citește de la tastatură numerele <code>a</code> și <code>b</code>.</p>

<h1>Date de ieșire</h1>

<p>Programul va afișa pe ecran numărul <code>x</code>, reprezentând numărul de numere prețioase din intervalul dat.</p>

<h1>Restricții și precizări</h1>

<ul>
	<li><code>1 ≤ a,b ≤ 10000000000000000000</code></li>
</ul>

<pre class="EnlighterJSRAW" data-enlighter-language="cpp">#include <bits/stdc++.h>
using namespace std;
unsigned long long start, stop;
unsigned long long a, b, i, nr, p = 1;
bool prim(long long x)
{
    bool ok = true;
    long long i;
    if(x == 0 or x == 1) return false;
    for(i = 2; i * i <= x; i ++)
    if(x % i == 0)
    {
        ok = false;
        break;
    }
    return ok;
}

int main()
{
    cin >> a >> b;
    start = log2(a) + 1;
    stop = log2(b);
    for(i = 1; i <= start; i ++)
        p = p * 2;
    if(prim(start))
    {
        nr = nr + p - a;
    }
    for(i = start; i < stop; i ++)
        if(prim(i + 1))
        {
            nr = nr + p;
            p = p * 2;
        } else p = p * 2;
    if(prim(stop + 1)) nr = nr + b - p + 1;
    cout << nr << " ";
    return 0;
}</pre>
