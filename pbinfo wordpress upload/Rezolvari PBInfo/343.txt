<h1>Cerinţa</h1>

<p>Se dau <code>n</code> numere întregi. Calculaţi cel mai mic dintre cele <code>n</code> numere date.</p>

<h1>Date de intrare</h1>

<p>Programul citește de la tastatură numărul <code>n</code>, iar apoi <code>n</code> numere întregi, separate prin spaţii.</p>

<h1>Date de ieşire</h1>

<p>Programul afișează pe ecran numărul <code>MIN</code>, reprezentând cel mai mic dintre cele <code>n</code> numere date.</p>

<h1>Restricţii şi precizări</h1>

<ul>
	<li><code>1 ≤ n ≤ 1000</code></li>
	<li>cele <code>n</code> numere citite vor avea cel mult <code>9</code> cifre</li>
</ul>

<pre class="EnlighterJSRAW" data-enlighter-language="cpp">#include <fstream>
using namespace std;

ifstream cin("soarece1.in");
ofstream cout("soarece1.out");

int n , m , is , js , ib , jb , a[11][11] , gata;
char s;

const int di[] = {0 ,  0 , 1 , -1};
const int dj[] = {1 , -1 , 0 ,  0};

int inside(int i , int j)
{
    return i >= 1 &amp;&amp; i <= n &amp;&amp; j >= 1 &amp;&amp; j <= m;
}

void afisare()
{
    gata++;
    for(int i = 1 ; i <= n ; ++i)
    {
        for(int j = 1 ; j <= m ; ++j)
            if(a[i][j] == -1) cout << 0 << " ";
            else cout << a[i][j] << " ";
        cout << '\n';
    }
}
void back(int i , int j , int pas)
{
    for(int d = 0 ; d < 4 &amp;&amp; !gata; d++)
    {
        int inou = i + di[d];
        int jnou = j + dj[d];
        if(inside(inou , jnou) &amp;&amp; a[inou][jnou] == 0)
        {
            a[inou][jnou] = pas;
            if(inou == ib &amp;&amp; jnou == jb) afisare();
            else back(inou , jnou , pas + 1);
            a[inou][jnou] = 0;
        }
    }
}
int main()
{
    cin >> n >> m;
    for(int i = 1 ; i <= n ; ++i)
        for(int j = 1 ; j <= m ; ++j)
        {
            cin >> s;
            if(s == '_') a[i][j] = 0;
            else if(s == '#') a[i][j] = -1;
            else if(s == 'S') a[i][j] = 0 , is = i , js = j;
            else if(s == 'B') a[i][j] = 0 , ib = i , jb = j;
        }
    a[is][js] = 1;
    back(is, js, 2);
    if(gata == 0)
        for(int i = 1 ; i <= n ; ++i)
        {
            for(int j = 1 ; j <= m ; ++j)
                cout << 0 << " ";
            cout << '\n';
        }
    return 0;
