<pre class="EnlighterJSRAW" data-enlighter-language="cpp">#include <fstream>
using namespace std;

ifstream cin("mygo.in");
ofstream cout("mygo.out");

int A[11];
const int mod = 666013;

int fact(int n){
    int r = 1;
    for(int i = 1; i <= n; ++i)
        r = 1LL * r * i % mod;
    return r;
}

int imod(int a){
    int n = mod - 2;
    int r = 1;
    while(n > 0){
        if(n % 2 == 1)
            r = 1LL * r * a % mod;
        a = 1ll * a * a % mod;
        n/=2;
    }
    return r;
}

int main(){
    int p = -1;
    for(int i = 0; i < 10; ++i)
        cin >> A[i], p += A[i];
    int s = 0;
    for(int k = 1; k < 10; ++k)
    if(A[k] > 0){
        A[k]--;
        int Q = 1;
        for(int i = 0; i <= 9; ++i)
            Q = 1LL * Q * fact(A[i]) % mod;
        s = (s + 1LL * fact(p) * imod(Q) % mod) % mod;
        A[k] ++;
    }
    cout << s;
    return 0;
}</pre>
