<h3>Explicație</h3>

<p>Primele <code>3</code> perechi satisfac condiția din enunț.</p>						
<pre class="EnlighterJSRAW" data-enlighter-language="cpp">#include <fstream>
using namespace std;

ifstream cin("fibo_gcd.in");
ofstream cout("fibo_gcd.out");

int n , x , y , cnt;

int prime_intre_ele(int a , int b)
{
    int r;
    while(b != 0)
    {
        r = a % b;
        a = b;
        b = r;
    }
    if(a == 1 || a == 2) return 1;
    else return 0;
}

int main()
{
    cin >> n;
    for(int i = 1 ; i <= n ; i++)
    {
        cin >> x >> y;
        if(prime_intre_ele(x , y)) cnt++;
    }
    cout << cnt;
    return 0;
}</pre>
