<p>Se consideră un șir <code>a[1]</code>, <code>a[2]</code>, …, <code>a[n]</code> de numere întregi.</p>

<h1>Cerința</h1>

<p>Să se determine diferența maximă de forma <code>a[i] - a[j]</code>, unde <code>1 ≤ i &lt; j ≤ n</code>.</p>

<h1>Date de intrare</h1>

<p>Programul citește de la tastatură numărul <code>n</code>, iar apoi șirul de <code>n</code> numere întregi, separate prin spații.</p>

<h1>Date de ieșire</h1>

<p>Programul va afișa pe ecran un singur număr întreg reprezentând diferența maximă cerută.</p>

<h1>Restricții și precizări</h1>

<ul>
	<li><code>1 ≤ n ≤ 100 000</code></li>
	<li><code>-1 000 000 000 ≤ a[i] ≤ 1 000 000 000</code></li>
</ul>

<pre class="EnlighterJSRAW" data-enlighter-language="cpp">#include <iostream>
#include <fstream>
#include <climits>
using namespace std;

ifstream fin("secvente.in");
ofstream fout("secvente.out");

int main()
{
    const int Inf = 0x3f3f3f3f;
    int a[100002];  
    int minim[100002];  
    int maxim[100002];  
    int n; 
    cin >> n;
    maxim[0] = -Inf;
    for (int i = 1; i <= n; ++i)
    {
        cin >> a[i];
        if (a[i] > maxim[i - 1])
            maxim[i] = a[i];
        else
            maxim[i] = maxim[i - 1];
    }
    
    minim[n + 1] = Inf;
    for (int i = n; i >= 1; --i)
        if (a[i] < minim[i + 1])
            minim[i] = a[i];
        else
            minim[i] = minim[i + 1];        
    int dif, dif_max = -Inf;
    for (int i = 1; i < n; ++i)
    {
        dif = maxim[i] - minim[i + 1];
        if (dif > dif_max)
            dif_max = dif;
    }
    cout << dif_max;
    
    fin.close();
    fout.close();
    
    return 0;
}</pre>
